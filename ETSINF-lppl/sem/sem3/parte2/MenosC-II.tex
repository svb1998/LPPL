\begin{traspa}{270}{Especificación Semántica}

\vspace{-5mm}
\begin{blistas}{0}{2}
\item Todas las variables y funciones deben declararse antes de ser
  utilizadas.
%
\item Debe haber una función, y solo una, con el nombre
  \texttt{main}. 
%   
\item La información de los parámetros se sitiuará en la TdS, en
  orden inverso \\[2mm]a su declaración.
%
\item El paso de parámetros se hace siempre por valor. 
%
\item Se admite la recursividad en las funciones.
%
\item En el compilador solo se usan constantes enteras. Si el
  analizador léxico encuentra\\[2mm]
  una constante real se debe devolver su valor entero truncado.
%
\item La talla de los tipos {\em entero} ~y {\em lógico} ~se debe
  definir en {\sc talla\_tipo\_simple}$=1$.
%
\item El tipo lógico \texttt{bol} se representa numéricamente como un
  entero: con el valor 0, \\[2mm]
  para el caso \texttt{falso}, y 1, para el caso \texttt{verdad}.
\end{blistas}

\end{traspa}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{traspa}{270}{Especificación Semántica}

\vspace{-5mm}
\begin{blistas}{0}{2}
%
\item No existe conversión de tipos entre ~\texttt{int} ~y
  ~\texttt{bool}.
%
\item Los índices de los vectores van de \texttt{0} a \texttt{cte-1},
  siendo \texttt{cte} el número de elementos, \\[2mm]
  que debe ser un entero positivo.
%
\item No es necesario comprobar los índices de los vectores en tiempo
  de ejecución.
%
\item Los operadores de incremento y decremento (infijos y postfijos)
  solo se pueden\\[2mm] aplicar a variables (p.ej. \texttt{(i+j)++}
  ~es ilegal).
%
\item En la instrucción \texttt{for} las \textit{expresiones
    opcionales} deben ser asignaciones o\\[2mm] o expresiones (pueden
  no aparecer). La \textit{expresión}, como en {\sc C}, debe ser
  \\[2mm]de tipo lógico y debe aparecer explícitamente.
%
\item La expresión de la instrucción \texttt{if-else} debe ser de
     tipo lógico.
%
\item Por defecto las restricciones semánticas serán las propias del
  lenguaje ANSI $C$.
\end{blistas}

\end{traspa}

\begin{traspa}{270}{Tabla de Símbolos}

\vspace{-10mm}
\begin{blistas}{0}{2}
\item \textcolor{blue}{\large Estructura de la TDS}

  Constantes, variables globales y estructuras
  básicas\hfill (ver Sección \texttt{5.1} del Enunciado)

\item \textcolor{blue}{\large Funciones de manipulación de la TDS}

\vspace{0mm}
\begin{small}
\begin{alltt}
void \underline{cargaContexto} (int n) ;
\textcolor{blue}{/* Crea el contexto necesario para los objetos globales y para los objetos 
   locales a las funciones                                                  */}
void \underline{descargaContexto} (int n) ;
\textcolor{blue}{/* Libera en la TdB y la TdS el contexto asociado con la función.           */}
int \underline{insTdS} (char *nom, int cat, int tipo, int n, int desp, int ref) ;
\textcolor{blue}{/* Inserta en la TdS toda la información asociada con una variable de nombre,
   "nom"; categoría, "cat"; tipo, "tipo"; nivel del bloque, "n"; desplaza-
   miento relativo, "desp"; y referencia, "ref", a posibles subtablas de 
   vectores o dominios, siendo (-1) si es de tipo simple. Si la variable ya 
   existe devuelve el valor "FALSE=0" ("TRUE=1" en caso contrario).         */}
SIMB \underline{obtTdS} (char *nom) ;
\textcolor{blue}{/* Obtiene toda la información asociada con un objeto de nombre, "nom", y la
   devuelve en una estructura de tipo "SIMB" (ver "libtds.h"). Si el objeto 
   no está declarado, devuelve "T_ERROR" en el campo "tipo".                */}
\end{alltt}
\end{small}
\end{blistas}

\end{traspa}

\begin{traspa}{270}{Tabla de Símbolos}

\vspace{-10mm}
\begin{blistas}{0}{-2}
\item[] \mbox{}
\begin{small}
\begin{alltt}
int \underline{insTdA} (int telem, int nelem) ;
\textcolor{blue}{/* Inserta en la Tabla de array la información de un array con elementos de 
   tipo, "telem"; y número de elementos, "nelem". Devuelve su referencia en 
   la Tabla de Arrays.                                                      */}
DIM \underline{obtTdA} (int ref) ;
\textcolor{blue}{/* Devuelve toda la información asociada con un array referenciado por "ref" 
   en la Tabla de Arrays. En caso de error devuelve "T_ERROR" en el campo 
   "telem".                                                                 */}
int \underline{insTdD} (int refe, int tipo) ;
\textcolor{blue}{/* Para un dominio existente referenciado por "refe", inserta en la Tabla 
   de Dominios la informacion del "tipo" del parámetro. Si "ref= -1" entonces
   crea una nueva entrada en la tabla de dominios para el tipo de este 
   parámetro y devuelve su referencia.  Si la funcion no tiene parametros, 
   debe crearse un dominio vacio con: "refe = -1" y "tipo = T_VACIO".       */}
INF \underline{obtTdD} (int refe) ;
\textcolor{blue}{/* Si "refe<0" entonces devuelve la informacion de la funcion actual, y si
   "refe>=0", devuelve la informacion de una función ya compilada con 
   referencia "refe". La informacion es: el nombre y el tipo del rango de la 
   función y la talla del segmento de parámetros. Si "refe" no se corresponde
   con una funcion ya compilada, devuelve "T_ERROR" en el campo "tipo".     */}

\end{alltt}
\end{small}
\end{blistas}

\end{traspa}

    {\em declaraciones}}

\vspace{0mm}
\begin{alltt}
declaracion | tipoSimple  ID_  AC_  CTE_  CC_  PCOMA_

            \{ int numelem = \$4; 
              if (\$4 <= 0)  \{
                yyerror("Talla inapropiada del array");
                numelem = 0;
              \}
              int refe = insTdA(\$1, numelem);
              if ( ! insTdS(\$2, VARIABLE, T_ARRAY, niv, dvar, refe) )
                yyerror ("Identificador repetido");  
             else dvar += numelem * TALLA_TIPO_SIMPLE;
            \}

\begin{traspa}{270}{ejemplo: comprobaciones semánticas}

\item \textcolor{blue}{\Large Ejemplo de comprobaciones de tipos en
    la {\em asignación}}

\vspace{-5mm}
\begin{alltt}
instruccionAsignacion
       | ID_  ASIG_  expresion  PCOMA_

         \{ SIMB sim = obtTdS(\$1);

           if (sim.t == T_ERROR) yyerror("Objeto no declarado");
           else if (! ((sim.t == \$3.t == T_ENTERO) ||
                       (sim.t == \$3-t == T_LOGICO)))
             yyerror("Error en la `instrucción de asignación'");
         \}
\end{alltt}
